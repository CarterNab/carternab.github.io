yalsP<- pitching %>% 
  filter(Tm =="Kansas City Royals") 
  averageERA<- pitching %>% 
    group_by(Season) %>% summarise(mean_ERA=mean(ERA))
ggplot() + 
  geom_line(data=pitching, aes(x=Season, y=ERA, group=Tm), color="light grey") +
  geom_line(data=yalsP, aes(x=Season, y=ERA), color="blue") +
  geom_line(data=averageERA, aes(x=Season, y=mean_ERA), color="black") +
  scale_y_continuous(limits = c(2, 6)) +
labs(
    x="Year", 
    y="ERA", 
    title="Royals Pitching Has Suffered Since Winning It All", 
    subtitle="KC's Average ERA isn't stacking up against the competition", 
    caption="Source: Sports-Reference | By Carter Nab"
    ) +
  theme_minimal() +
  theme(
      plot.title = element_text(size = 20, face = "bold"),
    axis.title = element_text(size = 14), 
    plot.subtitle = element_text(size=12), 
    panel.grid.minor = element_blank(),
    plot.title.position = "plot"
    )
