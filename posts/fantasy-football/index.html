library(tidyverse)
library(gt)
library(ggrepel)

seasontotals<- read_csv("SeasonTotals.csv") 

Top15PPR <- seasontotals %>% 
  top_n(15, wt=PPR)

Top15PPR %>% 
  select(Player, Tm, FantPos, PosRank, PPR) %>%
  arrange(desc(PPR)) %>% 
  gt() %>%
  cols_label(
    Tm = "Team",
    `FantPos` = "Position",
    `PosRank` = "Pos. Rank",
  ) %>%
  tab_header(
    title = "Do These Names Look Farmiliar?",
    subtitle = "They should! Through week 12 of this years NFL season these are the top 15 preforming players in Fantasy Football.") %>% 
  tab_style(
    style = cell_text(color = "black", weight = "bold", align = "left"),
    locations = cells_title("title")
  ) %>% tab_style(
    style = cell_text(color = "black", align = "left"),
    locations = cells_title("subtitle")
  ) %>%
  tab_source_note(
    source_note = md("By: Carter Nab  |  Source: Pro-Football-Reference")
  )

library(tidyverse)
library(ggrepel)
library(gt)

seasontotals <- read_csv("SeasonTotals.csv")

Touchdowns <- Top15PPR %>% 
  group_by(Player) %>%
  mutate(TotalTouchdowns=sum(RecTD, PassTD, RushTD))

ggplot() +
  geom_bar(data=Touchdowns, aes(x=reorder(Player, TotalTouchdowns), weight=TotalTouchdowns), fill= "red") +
  coord_flip() +
  labs(
    title = "Ever wondered who dominates the league in touchdowns?",
    subtitle = "No surprise that QB's throw a lot of touchdowns, did you expect to see a TE up there?",
    x = "",
    y = "Total Touchdowns",
    caption = "Pro-Football-Reference | By: Carter Nab"
  ) +
  theme_minimal() +
  theme(
    plot.title = element_text(size = 15, face = "bold"),
    axis.title = element_text(size = 10), 
    plot.subtitle = element_text(size=11), 
    panel.grid.minor = element_blank(),
    axis.text.x = element_text(size = 10),
    plot.title.position = "plot"
    )
